\section{Education}

\resumeSubHeadingListStart

<& for school in education | sort(attribute='startDate') | reverse &>

\resumeSubheading{%
  <&- if school.url &>
  \href{<<school.url>>}{<<school.institution>>}%
  <&- else &>
  <<school.institution>>%
  <&- endif &>
}{%
  <<school.location>>
}{%
  <&- if 'High School' is in(school.studyType) &>
  High School Diploma%
  <&- else &>
  <<school.studyType>>, <<school.area>>%
  <&- endif &>
}{%
  <<school.startDate | format_date>> -- <<school.endDate | default('') | format_date>>%
}

\vspace{-2pt}

\resumeItemListStart
<&- if school.expectedEndDate &>
\resumeItem{\textbf{Expected Graduation:} <<school.expectedEndDate | format_date(full_month=True)>>}
<&- endif &>

<&- if school.score &>
\resumeItem{\textbf{Score:} <<school.score>>}
<&- endif &>

\resumeItem{\textbf{Relevant Coursework:} %
<&- for course in school.courses &>
<<course>><<';' if not loop.last else ''>> %
<&- endfor &>
}
\resumeItemListEnd

\vspace{-4pt}

<& endfor &>

\resumeSubHeadingListEnd
\vspace{-12pt}
